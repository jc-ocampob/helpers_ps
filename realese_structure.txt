
# 1. make code changes

# 2. bump version in pyproject.toml
# example: 0.1.0 -> 0.1.1

# 3. optional: build locally to validate packaging
python -m build

# 4. commit
git add .
git commit -m "Release v0.2.1"

# 5. push branch
git push origin main

# 6. create tag
git tag v0.2.1
git push origin v0.2.1


# 7. Para instalaciones hacer lo siguiente reemplazando por la version v*
pip install "helpers_ps @ git+https://github.com/jc-ocampob/helpers_ps.git@v0.2.1"

# Para instalaciones ignorando las dependencias del paquete
pip install --no-deps "helpers_ps @ git+https://github.com/jc-ocampob/helpers_ps.git@v0.2.1"

# si es que la paqueteria ya esta instalada entonces usar la funcion de aquí
pip install --upgrade "helpers_ps @ git+https://github.com/jc-ocampob/helpers_ps.git@v0.2.1"
